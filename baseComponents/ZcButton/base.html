# button按钮组件(内置防抖功能)

防抖开启
0

在组件中需配置：

继承el-button所有属性

新增time属性（多少时间内点击；默认 .5 秒）

<template>
  <zc-layout-page>
    <zc-layout-page-item>
      <el-switch
        v-model="btnTime"
        :active-text="btnTime ? '防抖开启' : '防抖关闭'"
      >
      </el-switch>
      <zc-layout-page-item>
        <span>{{ num }}</span>
      </zc-layout-page-item>
    </zc-layout-page-item>
    <zc-layout-page-item>
      <zc-button
        :time="btnTime ? 500 : 0"
        size="small"
        type="primary"
        @click="exportExcel"
        >导出</zc-button
      >
    </zc-layout-page-item>
  </zc-layout-page>
</template>
<script>
export default {
  name: "buttonTime",
  data() {
    return {
      num: 0,
      btnTime: true,
    };
  },
  computed: {},
  created() {},
  methods: {
    // 导出
    async exportExcel() {
      this.num++;
      console.log("导出");
    },
  },
};
</script>
