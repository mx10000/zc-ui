# ZcReason原因分析组件

展示
这是原因分析组件 

<template>
  <zc-layout-page>
    <zc-layout-page-item>
  <el-switch
    v-model="active"
    :active-text="active ? '编辑' : '展示'"
  >
  </el-switch>
  <zc-reason :is-edit="active" :reason="reason" @input="inputChange" />
    </zc-layout-page-item>
  </zc-layout-page>
</template>
<script setup>

import { ref } from "vue";
const active=ref(false)
const reason=ref("这是原因分析组件")
const inputChange=(val)=>{
  reason.value=val
  console.log('正在输入',val)
}
</script>
