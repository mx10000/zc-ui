# tabs组件

# 默认

在组件中需配置

新增borderBlock或borderBlock=false属性

<template>
  <zc-layout-page>
    <zc-layout-page-item>
      <zc-tabs
        :border-block="false"
        :tabsList="tabsList"
        v-model="activeName"
        @tab-click="handleClick"
      >
      </zc-tabs>
    </zc-layout-page-item>
    <zc-layout-page-item>
      <zc-tabs
        :tabsList="tabsList"
        v-model="activeName"
        @tab-click="handleClick"
      >
      </zc-tabs>
    </zc-layout-page-item>
  </zc-layout-page>
</template>
<script>
export default {
  name: "index",
  data() {
    return {
      activeName: "first1",
      tabsList: [
        { name: "first", label: "测测测" },
        { name: "first1", label: "测测测" },
        { name: "first2", label: "测测测" },
        { name: "first3", label: "测测测" },
        { name: "first4", label: "测测测" },
        { name: "first6", label: "测测测" },
        { name: "first7", label: "测测测" },
        { name: "first8", label: "测测测" },
        { name: "first9", label: "测测测" },
        { name: "first10", label: "测测测" },
      ],
    };
  },
  methods: {
    handleClick(val) {
      console.log(val);
    },
  },
};
</script>

# 配置参数

参数 说明 类型 可选值 默认值
v-model 绑定值 boolean / string / number/Array - -
tabsList 数据源 Array - -
type 风格类型 string card/border-card -
size 大小 string small(仅在type=border-card时有效) -
tab-click tab 被选中时触发 - -
